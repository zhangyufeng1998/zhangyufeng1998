                                # vector

                                • 变长数组，倍增的思想
                                  • 系统为某一程序分配空间时，所需时间与空间大小无关，与申请次数有关
                                  • 每一次容量不够的时候，就把原来的容量乘2，然后把原来的数据复制过来
                                • size() 返回元素个数
                                • empty() 返回是否为空
                                • clear() 清空
                                • print()/back()
                                • push_back()/pop_back()
                                • begin()/end()
                                #include <vector>
using namespace std;
int main()
{
    vector<int> a(10,3);// 定义长度为10的vector 每个元素为3
}

                                #include <iostream>
#include <vector>

using namespace std;

int main() {
    vector<int> a;

    for (int i = 0; i < 10; i++) a.push_back(i);

    for (int i = 0; i < a.size(); i++) cout << a[i] << ' ';

    cout << endl;
    // a.begin() = a[0]
    // a.end() = a[a.size()]
    // vector<int>::iterator 可以替换成auto
    for (vector<int>::iterator i = a.begin(); i != a.end(); i++)
        cout << *i << endl;

    // auto可以自动推断变量类型
    for (auto x: a) cout << x << " ";
    cout << endl;
}

                                # pair<int,int>

                                • 存储一个二元组
                                • first，第一个元素
                                • second，第二个元素
                                • 支持比较运算，按字典序
                                #include <iostream>
#include <vector>
using namespace std;
int main() {
    pair<int, string> p;
    pair<int, string> s;

    // 初始化
    p = make_pair(10, "zyf");
    s = {20, "zyf"};
    cout << p.first << endl;
    cout << p.second << endl;
    
    // 使用piar存储三个不同的东西
    pair<int, pair<int, int>> z;
    z = {10, {20, 30}};
    cout << z.first << endl;
    cout << z.second.first << " " << z.second.second << endl;
    return 0;
}

                                # string

                                • 字符串
                                • substr(),c_str()
                                • size() 返回字母个数
                                • empty() 返回是否为空
                                • clear() 清空
                                #include <iostream>
#include <vector>

using namespace std;

int main() {
    string a = "zyf";
    a += "stc";
    a += '!';
    cout << a << endl;

    // substr第一个参数为起始位置，第二个参数为中止位置
    // 第二个参数省略就是从第一个参数到最后
    cout << a.substr(1, 2) << endl;

    printf("%s\n", a.c_str());
}

                                # queue

                                • 队列
                                • push(),对位插入一个元素
                                • front(),返回队头元素
                                • back(),返回队尾元素
                                • pop,弹出队头元素
                                • size(),empty()
                                • 没有clear()
                                #include <iostream>
#include <queue>

using namespace std;

int main() {
    queue<int> q;
    // 清空q
    q = queue<int>();
}

                                # deque

                                • 加强vector
                                • 双端队列，支持随机访问
                                • empty()
                                • size()
                                • clear()
                                • front()返回第一个元素
                                • back()返回最后一个元素
                                • push_back()/pop_back()
                                • push_front()/pop_front
                                • 支持随机寻址
                                • begin()/end()

                                # set,map,multiset,mutimap

                                • 基于平衡二叉树（红黑树）实现，动态维护有序序列
                                • size()
                                • empty()
                                • clear()
                                • begin() end() ++,-- 返回前驱和后继
                                • set multiset
                                  • insert() 插入一个数
                                  • find() 查找一个数
                                  • count() 返回某一个数的个数
                                  • earse()
                                    • 输入一个数x，删除所有x O（k+logn）k为x的个数
                                    • 输入一个迭代器，删除这个迭代器
                                  • lower_bound()/upper_bound()
                                    • lower_bound(x) 返回大于等于x的最小数的迭代器
                                    • upper_bound(x) 返回大于x的最小的数的迭代器
                                int main() {
    // 不可以有重复元素
    set<int> S;
    // 可以有重复元素
    multiset<int> MS;
}
                                • map multimap

                                  • insert() 插入的数是一个pair

                                  • erase() 输入的参数是pair或迭代器

                                  • find()

                                  • [] 时间复杂的是O（logn）

                                    • int main() {
map<string,int> a;
a["zyf"] = 1;
cout << a["zyf"] <<endl;
return 0;
}

                                  • lower_bound()/upper_bound()

                                # unordered_set,unordered_map,unordered_multiset,unordered_multimap

                                • 基于哈希表实现
                                • 和上面类似，增删改查的时间复杂度是O（1）
                                • 缺点是不支持lower_bound()/upper_bound()
